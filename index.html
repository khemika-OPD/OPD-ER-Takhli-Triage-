function triage() { // 1. ตรวจสอบการเลือก (ระดับ 1 และ 2) const isLevel1 = Array.from(document.querySelectorAll('.lv1')).some(c => c.checked); const isLevel2 = Array.from(document.querySelectorAll('.lv2')).some(c => c.checked); // 2. ดึงค่า V/S และ Lab (ระดับ 3) const age = document.getElementById('age').value; const hr = parseInt(document.getElementById('hr').value) || 0; const rr = parseInt(document.getElementById('rr').value) || 0; const sbp = parseInt(document.getElementById('sbp').value) || 0; const temp = parseFloat(document.getElementById('temp').value) || 0; const spo2 = parseInt(document.getElementById('spo2').value) || 100; const fbs = parseInt(document.getElementById('lab_fbs').value) || 100; const na = parseInt(document.getElementById('lab_na').value) || 140; const k = parseFloat(document.getElementById('lab_k').value) || 4.0; const wbc = parseInt(document.getElementById('lab_wbc').value) || 5000; const vct = parseInt(document.getElementById('lab_vct').value) || 10; // 3. ตรวจสอบเงื่อนไขระดับ 3 (เทียบค่าปกติ) let isLevel3 = false; // เช็ค V/S ผิดปกติแต่ไม่มีอาการ (กล่องสีส้มในผัง) if (hr >= 120 || hr < 50 || sbp >= 160 || temp >= 38) isLevel3 = true; // เช็ค Lab วิกฤต (Panic Value) if (fbs <= 55 || fbs >= 400) isLevel3 = true; if (na <= 120 || na >= 160) isLevel3 = true; if (k <= 2.5 || k >= 6.0) isLevel3 = true; if (wbc <= 2000 || wbc >= 30000) isLevel3 = true; if (vct >= 20) isLevel3 = true; // VCT > 20 นาที // 4. ตัดสินผลตามลำดับความรุนแรง (Hierarchy) let final; if (isLevel1 || spo2 < 90) { final = { title: 'ระดับ 1 (Resuscitation)', color: '#ffcccc', text: '#b30000', dest: 'ห้องฉุกเฉิน (ทันที)' }; } else if (isLevel2 || spo2 < 95 || checkVsByAge(age, hr, rr)) { final = { title: 'ระดับ 2 (Emergency)', color: '#ffe6f2', text: '#cc0066', dest: 'ห้องฉุกเฉิน (ด่วน)' }; } else if (isLevel3) { final = { title: 'ระดับ 3 (Urgent)', color: '#ffffcc', text: '#856404', dest: 'OPD Fast Track' }; } else { final = { title: 'ระดับ 4-5', color: '#ffffff', text: '#333', dest: 'ตรวจตามคิวปกติ' }; } renderResult(final); } // ฟังก์ชันแยกเช็ค V/S ตามช่วงอายุ (เกณฑ์ระดับ 2) function checkVsByAge(age, hr, rr) { if (age === 'adult' && (hr > 100 || rr > 20)) return true; if (age === 'child_8' && (hr > 140 || rr > 30)) return true; if (age === 'child_3' && (hr > 160 || rr > 40)) return true; if (age === 'infant' && (hr > 180 || rr > 50)) return true; return false; }
